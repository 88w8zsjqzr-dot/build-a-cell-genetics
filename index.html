    // ---------- Genetics engine (keep this simple to avoid science mistakes) ----------

    // Internal allele codes:

    // IA = Iᴬ, IB = Iᴮ, i = i

    function genotypeToMarkers(a1, a2) {

      const alleles = [a1, a2];

      const hasIA = alleles.includes("IA");

      const hasIB = alleles.includes("IB");

      return { A: hasIA, B: hasIB };

    }


    function markersToPhenotype(markers) {

      if (markers.A && markers.B) return "AB";

      if (markers.A) return "A";

      if (markers.B) return "B";

      return "O";

    }


    function genotypeToPhenotype(a1, a2) {

      return markersToPhenotype(genotypeToMarkers(a1, a2));

    }


    function formatAllele(code) {

      if (code === "IA") return "Iᴬ";

      if (code === "IB") return "Iᴮ";

      if (code === "i") return "i";

      return "___";

    }


    function formatGenotype(a1, a2) {

      if (!a1 || !a2) return "___";

      // Sort for consistent display: IA, IB, i

      const order = { "IA": 0, "IB": 1, "i": 2 };

      const sorted = [a1, a2].sort((x, y) => order[x] - order[y]);

      return formatAllele(sorted[0]) + formatAllele(sorted[1]);

    }

    // ---------- Checking logic + feedback (specific, not vague) ----------

    function buildReason(a1, a2, correct) {

      // Provide short, accurate reasons based on genotype

      const sorted = [a1, a2].slice().sort((x,y) => ({"IA":0,"IB":1,"i":2}[x]-{"IA":0,"IB":1,"i":2}[y]));

      const g = formatGenotype(sorted[0], sorted[1]);


      if (correct === "AB") {

        return g + " is AB because Iᴬ and Iᴮ are co-dominant, so both A and B markers appear.";

      }

      if (correct === "A") {

        if (sorted.includes("IA") && sorted.includes("i")) {

          return g + " is A because Iᴬ dominates i, so A markers appear.";

        }

        return g + " is A because the genotype contains Iᴬ, so A markers appear.";

      }

      if (correct === "B") {

        if (sorted.includes("IB") && sorted.includes("i")) {

          return g + " is B because Iᴮ dominates i, so B markers appear.";

        }

        return g + " is B because the genotype contains Iᴮ, so B markers appear.";

      }

      // O

      return g + " is O because ii produces no A or B markers.";

    }


    function checkAnswer() {

      if (!state.slot1 || !state.slot2) {

        setFeedback("bad", "You need two alleles. Drag one allele into each slot.");

        return;

      }

      if (!state.selectedPhenotype) {

        setFeedback("bad", "Choose a phenotype (A, B, AB, or O) before checking.");

        return;

      }


      const correct = genotypeToPhenotype(state.slot1, state.slot2);

      const chosen = state.selectedPhenotype;


      if (chosen === correct) {

        setFeedback("ok", "Correct. " + buildReason(state.slot1, state.slot2, correct));

        return;

      }


      // Wrong answer: give a precise correction and a targeted misconception fix

      const correctReason = buildReason(state.slot1, state.slot2, correct);


      let misconception = "";

      if (chosen === "O" && correct !== "O") {

        misconception = " O requires ii (two recessive alleles). Your genotype contains a dominant allele that produces a marker.";

      } else if (chosen === "AB" && correct !== "AB") {

        misconception = " AB requires both Iᴬ and Iᴮ together.";

      } else if ((chosen === "A" || chosen === "B") && correct === "AB") {

        misconception = " Because both Iᴬ and Iᴮ are present, both markers appear (co-dominance).";

      }


      setFeedback(

        "bad",

        "Not correct. Correct phenotype is " + correct + ". " + correctReason + misconception

      );

    }


    checkBtn.addEventListener("click", checkAnswer);


    resetBtn.addEventListener("click", () => {

      state.slot1 = null;

      state.slot2 = null;

      resetSelection();

      updateUI();

      setFeedback("warn", "Reset complete. Drag two alleles into the slots.");

    });


    // New round: random genotype challenge (still within the rules)

    // This keeps it game-like and replayable.

    function randomGenotype() {

      const alleles = ["IA", "IB", "i"];

      const a1 = alleles[Math.floor(Math.random() * alleles.length)];

      const a2 = alleles[Math.floor(Math.random() * alleles.length)];

      return [a1, a2];

    }


    newBtn.addEventListener("click", () => {

      const [a1, a2] = randomGenotype();

      state.slot1 = a1;

      state.slot2 = a2;

      resetSelection();

      updateUI();

      setFeedback("warn", "New round loaded. Choose the phenotype, then click Check answer.");

    });


    // Initial render

    updateUI();
