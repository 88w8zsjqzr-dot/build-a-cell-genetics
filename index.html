  <script>

    const slots = [null, null];

    const s1 = document.getElementById("s1");

    const s2 = document.getElementById("s2");

    const rbc = document.getElementById("rbc");

    const fb = document.getElementById("fb");


    function updateCell() {

      rbc.innerHTML = "";

      if (slots.includes("IA")) addMarkers("A");

      if (slots.includes("IB")) addMarkers("B");

    }


    function addMarkers(type) {

      for (let i = 0; i < 6; i++) {

        const m = document.createElement("div");

        m.className = "marker " + type;

        m.style.left = Math.random() * 200 + "px";

        m.style.top = Math.random() * 200 + "px";

        rbc.appendChild(m);

      }

    }


    document.querySelectorAll(".allele").forEach(a => {

      a.addEventListener("dragstart", e => {

        e.dataTransfer.setData("allele", a.dataset.a);

      });

    });


    [s1, s2].forEach((slot, i) => {

      slot.addEventListener("dragover", e => e.preventDefault());

      slot.addEventListener("drop", e => {

        const al = e.dataTransfer.getData("allele");

        slots[i] = al;

        slot.textContent = al;

        updateCell();

      });

    });


    document.querySelectorAll(".pick").forEach(p => {

      p.addEventListener("click", () => {

        document.querySelectorAll(".pick").forEach(x => x.classList.remove("selected"));

        p.classList.add("selected");


        if (!slots[0] || !slots[1]) {

          fb.textContent = "You need two alleles first.";

          return;

        }


        const hasA = slots.includes("IA");

        const hasB = slots.includes("IB");

        let correct = "O";

        if (hasA && hasB) correct = "AB";

        else if (hasA) correct = "A";

        else if (hasB) correct = "B";


        fb.textContent = (p.dataset.p === correct)

          ? "Correct ✔"

          : "Not correct. Correct blood group is " + correct;

      });

    });

  </script>
